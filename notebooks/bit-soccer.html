def gtc(N,Q,p_max,sample_number,p_min=1,g_min=None,g_max=None):
    if not g_min:
        g_min = p_min
    if not g_max:
        g_max = p_max
    P = rsg(p_min,p_max+1,N)
    G = rsg(g_min,g_max+1,Q)

    fob = open('test/sample-%d.in'%sample_number,'w')
    fob.write('%d\n%d'%(N,P.next()))
    for p in P:
        fob.write(' %d'%p)
    fob.write('\n%d\n'%Q)
    for q in G:
        fob.write('%d\n'%q)
    fob.close()

def do(P,Q):
    P,Q=map(np.array,[P,Q])
    for q in Q:
        zz=_not((~P&~q)^(~q))
        xx=_and(zz,P)
        dd=reduce(lambda x,y:x|y,xx)

#!/home/amila/tensorflow/bin python2
import numpy as np

_and=np.vectorize(lambda condition,value:
                  value if condition else 0)

_not=np.vectorize(lambda value:not value)

def do(P,Q):
    P,Q=map(np.array,[P,Q])
    for q in Q:
        zz=_not((~P&~q)^(~q))
        xx=_and(zz,P)
        dd=reduce(lambda x,y:x|y,xx)
        if dd==q:
            print 'YES'
        else:
            print 'NO'

N=input()
P=map(int, raw_input().split())
t=input()
Q=[input() for _ in range(t)]
do(P,Q)

%%file sol2.py

import numpy as np

_and=np.vectorize(lambda condition,value:
                  value if condition else 0)

_not=np.vectorize(lambda value:not value)

def do(P,Q):
    P,Q=map(np.array,[P,Q])
    for q in Q:
        zz=_not((~P&~q)^(~q))
        xx=_and(zz,P)
        dd=0
        for x in xx:
            dd=x|dd
            if dd==q:
                print 'YES'
                break
        else:
            print 'NO'

N=input()
P=map(int, raw_input().split())
t=input()
Q=[input() for _ in range(t)]
do(P,Q)

%%file sol3.py

import numpy as np

def do(P,Q):
    P,Q=map(np.array,[P,Q])
    for q in Q:
        zz=((~P&~q)^(~q))==0
        xx=P[zz]
        dd=0
        for x in xx:
            dd=x|dd
            if dd==q:
                print 'YES'
                break
        else:
            print 'NO'
